// src/risk.ts - 修改 parseRiskArgument 函数

function parseRiskArgument(content: string, role: RiskArgument["role"]): RiskArgument {
  const verdictMatch = content.match(/verdict[：:*]+\s*(pass|revise|reject)/i) ||
                       content.match(/结论[：:*]+\s*(pass|revise|reject|通过|修订|拒绝)/i);
  let verdict: RiskArgument["verdict"] = "pass";
  if (verdictMatch) {
    const raw = verdictMatch[1].toLowerCase();
    if (raw === "revise" || raw === "修订") verdict = "revise";
    else if (raw === "reject" || raw === "拒绝") verdict = "reject";
  }

  // 支持 ## 和 ### 两种标题级别
  // evidence：按子标题分割（### 证据N：...），保留标题文字作为证据摘要
  const evidenceRegex = /#{2,3}\s*证据[一二三四五六七八九十\d]+[：:]\s*(.+?)(?=#{2,3}\s*证据|\n#{2,3}\s*\d|$)/gs;
  const evidenceMatches = content.match(evidenceRegex) || [];
  const evidence = evidenceMatches
    .map((m) => {
      // 去掉 ### 前缀但保留标题内容（标题本身就是证据摘要）
      return m.replace(/^#{2,3}\s*/, "").trim();
    })
    .filter((e) => e.length > 0);

  // 如果没有子标题格式，回退到 bullet list 格式
  if (evidence.length === 0) {
    const evidenceSection = content.match(/#{2,3}\s*2\.?\s*证据支撑\s*\n([\s\S]*?)(?=\n#{2,3}\s*\d|$)/);
    if (evidenceSection) {
      const bulletMatches = evidenceSection[1].match(/^- (.+)/gm) || [];
      evidence.push(...bulletMatches.map((m) => m.replace(/^- /, "").trim()));
    }
  }

  // 支持 ## 和 ### 两种标题级别，匹配立场声明到下一个标题为止
  const positionMatch = content.match(/#{2,3}\s*1\.?\s*立场声明[：:]?\s*\n([\s\S]*?)(?=\n#{2,3}\s*\d|$)/);

  return {
    role,
    position: positionMatch ? positionMatch[1].trim() : "",
    evidence,
    verdict,
  };
}
```

/**
 * 生成结构化的决策理由，解释为什么最终决策可能与某些分析师结论不一致
 */
function generateDecisionRationale(
  analystReports: AnalystReport[],
  debate: DebateResult,
  researchDecision: ResearchDecision,
  riskAssessment: RiskAssessment
): string {
  const parts: string[] = [];
  
  // 1. 分析师共识（复用 analystVerdictSummary 的逻辑）
  const summary = analystVerdictSummary(analystReports);
  parts.push(`分析师共识：${summary}`);
  
  // 2. 辩论结果
  if (debate.rounds.length > 0) {
    parts.push(`多空辩论：Bull ${researchDecision.bull_score} vs Bear ${researchDecision.bear_score}`);
  }
  
  // 3. 风控状态
  if (riskAssessment.status !== "pass") {
    parts.push(`风控意见：${riskAssessment.status}`);
    if (riskAssessment.retries_exhausted) {
      parts.push(`（重试耗尽，保留原始决策）`);
    }
  }
  
  // 4. 技术面与基本面冲突检测
  const marketAnalyst = analystReports.find(r => r.role === "market");
  const fundamentalsAnalyst = analystReports.find(r => r.role === "fundamentals");
  if (marketAnalyst && fundamentalsAnalyst) {
    const mDir = marketAnalyst.verdict.direction;
    const fDir = fundamentalsAnalyst.verdict.direction;
    const isBearish = mDir === "看空" || mDir === "Sell" || mDir === "Underweight";
    const isBullish = fDir === "看多" || fDir === "Buy" || fDir === "Overweight";
    if (isBearish && isBullish) {
      parts.push(`技术面看空但基本面看多，采用左侧逆向策略`);
    }
  }
  
  return parts.join("；");
}
```

  bear_summary: string;
  convergence_score: number;  // 新增：收敛分数 0-1
  resolved_points: string[];  // 新增：已解决的论点摘要
  total_tokens: number;
  total_cost_usd: number;

```typescript
// src/debate.ts - 修改 runBullBearDebate 函数的 return 语句

// 在 return 前计算收敛分数
// 基于 rounds 数组中的数据，不需要新增全局变量
const totalUnresolved = rounds[rounds.length - 1]?.unresolved_ids?.length || 0;
const totalResolved = rounds.reduce((sum, r) => sum + (r.resolved_ids?.length || 0), 0);
const totalClaims = totalResolved + totalUnresolved;

// 收敛分数：已解决论点占比
const baseConvergence = totalClaims > 0 ? totalResolved / totalClaims : 0.5;

// 发散惩罚：最后轮 unresolved > 第一轮
let divergencePenalty = 0;
if (rounds.length >= 2) {
  const firstRoundUnresolved = rounds[0].unresolved_ids?.length || 0;
  const lastRoundUnresolved = rounds[rounds.length - 1].unresolved_ids?.length || 0;
  if (lastRoundUnresolved > firstRoundUnresolved) {
    divergencePenalty = Math.min(0.3, (lastRoundUnresolved - firstRoundUnresolved) * 0.1);
  }
}

const convergenceScore = Math.max(0, baseConvergence - divergencePenalty);

// 生成已解决论点摘要
const resolvedPoints = rounds
  .flatMap(r => r.resolved_ids || [])
  .map(id => {
    const claim = registry.byId.get(id);
    return claim ? `${claim.id}: ${claim.topic}` : id;
  });

return {
  rounds,
  bull_summary: bullSummary,
  bear_summary: bearSummary,
  convergence_score: convergenceScore,
  resolved_points: resolvedPoints,
  total_tokens: totalTokens,
  total_cost_usd: totalCostUsd,
};
```

if (riskAssessment.judge?.hard_constraints) {

export interface PipelineIssue {
  /** Pipeline stage where the issue was detected. */
  stage: "data_collection" | "template_render" | "analyst_output" | "quality_gate" | "quality_review" | "debate" | "cross_stage";
  /** abort = stop pipeline; skip = skip this item; warn = record only. */

  /** Short check name (e.g. "placeholders_remaining"). */
  check: string;
  /** Human-readable description. */
  message: string;
  /** Optional context (role, placeholder names, etc.). */
  context?: Record<string, any>;
}
